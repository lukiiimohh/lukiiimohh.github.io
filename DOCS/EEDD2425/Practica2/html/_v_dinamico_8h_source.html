308int VDinamico<T>::busquedaBin(T& dato){
309 int inf = 0;
310 int sup = _tamal - 1;
311 int curIn;
312 while (inf <= sup) {
313 curIn = (inf + sup) / 2;
314 if (_datos[curIn] == dato) {
315 return curIn;
316 }else if (_datos[curIn] < dato) {
317 inf = curIn + 1;

319 sup = curIn - 1;
320 }
321 }
322 return -1;
